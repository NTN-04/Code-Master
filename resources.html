Xác Thực Form Tương Tác

HTML CSS JavaScript

Xác thực form thời gian thực với JavaScript và phản hồi trực quan. 

const form = document.getElementById('form');
const username = document.getElementById('username');
const email = document.getElementById('email');

form.addEventListener('submit', (e) => {
  if (!validateInputs()) {
    e.preventDefault();
  }
});

function validateInputs() {
  const usernameValue = username.value.trim();
  const emailValue = email.value.trim();
  let isValid = true;
  
  if (usernameValue === '') {
    setError(username, 'Tên người dùng là bắt buộc');
    isValid = false;
  } else {
    setSuccess(username);
  }
  
  if (emailValue === '') {
    setError(email, 'Email là bắt buộc');
    isValid = false;
  } else if (!isValidEmail(emailValue)) {
    setError(email, 'Vui lòng cung cấp địa chỉ email hợp lệ');
    isValid = false;
  } else {
    setSuccess(email);
  }
  
  return isValid;
}

function App() {
  const [todos, setTodos] = useState([]);
  const [inputValue, setInputValue] = useState('');

  const addTodo = () => {
    if (inputValue.trim() !== '') {
      setTodos([
        ...todos, 
        { 
          id: Date.now(), 
          text: inputValue, 
          completed: false 
        }
      ]);
      setInputValue('');
    }
  };

  const toggleTodo = (id) => {
    setTodos(
      todos.map(todo => 
        todo.id === id ? 
        { ...todo, completed: !todo.completed } : 
        todo
      )
    );
  };

  return (
    <div className="app">
      <h1>Ứng Dụng Todo</h1>

    </div>
  );
}
